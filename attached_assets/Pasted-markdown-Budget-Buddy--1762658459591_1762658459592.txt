```typescript
import { Router } from 'express';
import { db } from '../db/client';
import { transactions } from '../db/schema';

import { eq } from 'drizzle-orm';

const router = Router();

// –ü–æ–ª—É—á–∏—Ç—å –≤—Å–µ —Ç—Ä–∞–Ω–∑–∞–∫—Ü–∏–∏ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è

  const userId = req.user.id;
  
  const data = await db
    .select()
    .from(transactions)
    .where(eq(transactions.userId, userId))
    .orderBy(transactions.date);
  
  res.json(data);
});

// –°–æ–∑–¥–∞—Ç—å —Ç—Ä–∞–Ω–∑–∞–∫—Ü–∏—é
router.post('/', requireAuth, async (req, res) => {
  const userId = req.user.id;
  const { date, type, amount, description, category } = req.body;
  
  const [transaction] = await db
    .insert(transactions)
    .values({
      userId,
      date,
      type,
      amount,
      amountUsd: amount, // TODO: currency conversion
      description,
      category
    })
    .returning();
  
  res.json(transaction);
});

export default router;
```

```typescript
// client/src/hooks/useTransactions.ts

import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { api } from '../lib/api';

export function useTransactions() {
  const queryClient = useQueryClient();
  
  // –ü–æ–ª—É—á–∏—Ç—å –≤—Å–µ —Ç—Ä–∞–Ω–∑–∞–∫—Ü–∏–∏
  const { data: transactions = [] } = useQuery({
    queryKey: ['transactions'],
    queryFn: () => api.get('/transactions')
  });
  
  // –°–æ–∑–¥–∞—Ç—å —Ç—Ä–∞–Ω–∑–∞–∫—Ü–∏—é
  const createMutation = useMutation({
    mutationFn: (data) => api.post('/transactions', data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['transactions'] });
    }
  });
  
  // –£–¥–∞–ª–∏—Ç—å —Ç—Ä–∞–Ω–∑–∞–∫—Ü–∏—é
  const deleteMutation = useMutation({
    mutationFn: (id) => api.delete(`/transactions/${id}`),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['transactions'] });
    }
  });
  
  return {
    transactions,
    create: createMutation.mutate,
    delete: deleteMutation.mutate,
    isLoading: createMutation.isPending || deleteMutation.isPending
  };
}
```
