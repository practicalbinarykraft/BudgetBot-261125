    id: number;

    amount: number;
    icon?: string;
    priority: number;
    prediction?: {
      monthsToAfford: number | null;

export function GoalTimelineMarker({
  goal,
  currentMonth,
  onHover,
  onClick
}: GoalTimelineMarkerProps) {
  if (!goal.prediction?.monthsToAfford) {
    return null; // –ù–µ –ø–æ–∫–∞–∑—ã–≤–∞—Ç—å –µ—Å–ª–∏ –Ω–µ –º–æ–∂–µ–º –ø—Ä–µ–¥—Å–∫–∞–∑–∞—Ç—å

  // –¶–≤–µ—Ç –ø–æ –ø—Ä–∏–æ—Ä–∏—Ç–µ—Ç—É
  const color = goal.priority === 1 ? 'red'

  const size = goal.priority === 1 ? 'large'

  return (

      style={{
