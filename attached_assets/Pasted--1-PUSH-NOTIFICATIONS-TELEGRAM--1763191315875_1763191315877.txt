export async function getTodayPlannedExpenses(userId: number) {
  const today = new Date();
  const todayStr = today.toISOString().split('T')[0]; // "2025-11-15"
  
  const expenses = await db
    .select()

      )
    );
  
  const total = expenses.reduce((sum, e) => sum + parseFloat(e.amount), 0);

    total,

/**
 * –ü–æ–ª—É—á–∏—Ç—å —Ä–∞—Å—Ö–æ–¥—ã –Ω–∞ –Ω–µ–¥–µ–ª—é
 * targetDate –º–µ–∂–¥—É —Å–µ–≥–æ–¥–Ω—è –∏ +7 –¥–Ω–µ–π
 */
export async function getWeekPlannedExpenses(userId: number) {
  const today = new Date();
  const nextWeek = new Date(today);
  nextWeek.setDate(nextWeek.getDate() + 7);
  
  const todayStr = today.toISOString().split('T')[0];
  const nextWeekStr = nextWeek.toISOString().split('T')[0];
  
  const expenses = await db
    .select()
    .from(futurePurchases)
    .where(
      and(
        eq(futurePurchases.userId, userId),
        gte(futurePurchases.targetDate, todayStr),
        lte(futurePurchases.targetDate, nextWeekStr),
        eq(futurePurchases.isPurchased, false)
      )
    )
    .orderBy(futurePurchases.targetDate);
  
  // –ì—Ä—É–ø–ø–∏—Ä–æ–≤–∞—Ç—å –ø–æ –¥–Ω—è–º
  const byDay = expenses.reduce((acc, expense) => {
    const date = expense.targetDate!;
    if (!acc[date]) {
      acc[date] = [];
    }
    acc[date].push(expense);
    return acc;
  }, {} as Record<string, typeof expenses>);
  
  const total = expenses.reduce((sum, e) => sum + parseFloat(e.amount), 0);
  
  return {
    expenses,
    byDay,
    total,
    count: expenses.length
  };
}

export async function canAffordTodayExpenses(userId: number) {
  const { total } = await getTodayPlannedExpenses(userId);
  
  // –ü–æ–ª—É—á–∏—Ç—å —Å–≤–æ–±–æ–¥–Ω—ã–π –∫–∞–ø–∏—Ç–∞–ª
  const stats = await getMonthlyStats(userId);

  return {
    canAfford: freeCapital >= total,
    total,
    freeCapital,
    deficit: Math.max(0, total - freeCapital)
  };
}

  const insights: string[] = [];
  
  // –ò–Ω—Å–∞–π—Ç 1: –ü—Ä–µ–≤—ã—à–µ–Ω–∏–µ –±—é–¥–∂–µ—Ç–∞

  if (summaryData.goalProgress) {

        `${delayedGoals[0].name} will be delayed if you continue overspending.`

        `Cut ${highestOverspend.category} spending by 50% to get back on track!`
      );
    }
  }
  
  return insights;

function formatDayName(dateStr: string): string {
  const date = new Date(dateStr);
  const days = ['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'];
  const day = days[date.getDay()];
  const dayNum = date.getDate();
  return `${day} ${dayNum}`;
}
