import { useState, useEffect } from 'react';
import { useMutation, useQuery } from '@tanstack/react-query';
import { useTranslation } from 'react-i18next';
import { X, Loader2, ExternalLink, TrendingDown, CheckCircle } from 'lucide-react';

interface SearchResult {
  store: string;
  price: number;
  url: string;
  availability: string;
}

interface SearchReport {
  id: number;
  status: string;
  results: SearchResult[];
  bestPrice: string;
  bestStore: string;
  savings: string;
  savingsPercent: string;
}

interface Props {
  productId: number;
  productName: string;
  onClose: () => void;
}

export function PriceSearchModal({ productId, productName, onClose }: Props) {
  const { t } = useTranslation();
  const [searchStarted, setSearchStarted] = useState(false);
  
  // –ó–∞–ø—É—Å—Ç–∏—Ç—å –ø–æ–∏—Å–∫
  const searchMutation = useMutation({
    mutationFn: async () => {
      const response = await fetch(`/api/product-catalog/${productId}/search-prices`, {
        method: 'POST'
      });
      if (!response.ok) throw new Error('Search failed');
      return response.json();
    },
    onSuccess: () => {
      setSearchStarted(true);
    }
  });
  
  // –ü–æ–ª—É—á–∏—Ç—å –æ—Ç—á—ë—Ç (–µ—Å–ª–∏ –µ—Å—Ç—å)
  const { data: reportData } = useQuery({
    queryKey: [`/api/product-catalog/${productId}/search-report`],
    enabled: searchStarted,
    refetchInterval: searchStarted ? 2000 : false // Poll –∫–∞–∂–¥—ã–µ 2 —Å–µ–∫
  });
  
  const report = reportData as SearchReport | undefined;
  
  const handleStartSearch = () => {
    searchMutation.mutate();
  };
  
  const isSearching = searchMutation.isPending || 
                     (report?.status === 'pending');
  const isCompleted = report?.status === 'completed';
  
  return (
    <div className="fixed inset-0 bg-black bg-opacity-50 flex items-center justify-center z-50 p-4">
      <div className="bg-white rounded-lg max-w-2xl w-full max-h-[80vh] overflow-y-auto">
        {/* Header */}
        <div className="p-6 border-b flex items-start justify-between">
          <div>
            <h2 className="text-xl font-bold mb-1">
              {t('priceSearch.title', '–ü–æ–∏—Å–∫ –ª—É—á—à–µ–π —Ü–µ–Ω—ã')}
            </h2>
            <p className="text-sm text-gray-600">{productName}</p>
          </div>
          <button
            onClick={onClose}
            className="p-1 hover:bg-gray-100 rounded"
          >
            <X className="w-5 h-5" />
          </button>
        </div>
        
        {/* Content */}
        <div className="p-6">
          {/* –ö–Ω–æ–ø–∫–∞ –∑–∞–ø—É—Å–∫–∞ –ø–æ–∏—Å–∫–∞ */}
          {!searchStarted && !isSearching && (
            <div className="text-center py-8">
              <p className="text-gray-600 mb-4">
                {t('priceSearch.description', 'AI –ø–æ–∏—â–µ—Ç –∞–∫—Ç—É–∞–ª—å–Ω—ã–µ —Ü–µ–Ω—ã –≤ –∏–Ω—Ç–µ—Ä–Ω–µ—Ç-–º–∞–≥–∞–∑–∏–Ω–∞—Ö')}
              </p>
              <button
                onClick={handleStartSearch}
                disabled={searchMutation.isPending}
                className="px-6 py-3 bg-purple-500 text-white rounded-lg hover:bg-purple-600 transition disabled:opacity-50"
              >
                {t('priceSearch.start', '–ù–∞—á–∞—Ç—å –ø–æ–∏—Å–∫')}
              </button>
              <p className="text-xs text-gray-500 mt-2">
                {t('priceSearch.hint', '–ó–∞–π–º—ë—Ç 10-20 —Å–µ–∫—É–Ω–¥')}
              </p>
            </div>
          )}
          
          {/* Loading */}
          {isSearching && (
            <div className="text-center py-12">
              <Loader2 className="w-12 h-12 animate-spin text-purple-500 mx-auto mb-4" />
              <p className="text-gray-600">
                {t('priceSearch.searching', '–ò—â–µ–º –ª—É—á—à–∏–µ —Ü–µ–Ω—ã...')}
              </p>
              <p className="text-sm text-gray-500 mt-2">
                {t('priceSearch.pleaseWait', '–ü–æ–∂–∞–ª—É–π—Å—Ç–∞, –ø–æ–¥–æ–∂–¥–∏—Ç–µ')}
              </p>
            </div>
          )}
          
          {/* Results */}
          {isCompleted && report && (
            <div>
              {/* –≠–∫–æ–Ω–æ–º–∏—è */}
              {parseFloat(report.savings) > 0 && (
                <div className="mb-6 p-4 bg-green-50 border border-green-200 rounded-lg">
                  <div className="flex items-start gap-3">
                    <CheckCircle className="w-6 h-6 text-green-600 flex-shrink-0 mt-1" />
                    <div>
                      <p className="font-semibold text-green-900 mb-1">
                        {t('priceSearch.savingsFound', '–ù–∞–π–¥–µ–Ω–∞ —ç–∫–æ–Ω–æ–º–∏—è!')}
                      </p>
                      <p className="text-2xl font-bold text-green-700">
                        ${parseFloat(report.savings).toFixed(2)}
                      </p>
                      <p className="text-sm text-green-700">
                        ({parseFloat(report.savingsPercent).toFixed(1)}% –¥–µ—à–µ–≤–ª–µ)
                      </p>
                    </div>
                  </div>
                </div>
              )}
              
              {/* –°–ø–∏—Å–æ–∫ –º–∞–≥–∞–∑–∏–Ω–æ–≤ */}
              <div>
                <h3 className="font-semibold mb-3">
                  {t('priceSearch.results', '–†–µ–∑—É–ª—å—Ç–∞—Ç—ã')} ({report.results?.length || 0})
                </h3>
                <div className="space-y-3">
                  {report.results?.map((result, index) => (
                    <div 
                      key={index}
                      className={`p-4 border rounded-lg ${
                        result.store === report.bestStore 
                          ? 'border-green-500 bg-green-50' 
                          : ''
                      }`}
                    >
                      <div className="flex items-start justify-between mb-2">
                        <div>
                          <h4 className="font-semibold flex items-center gap-2">
                            {result.store}
                            {result.store === report.bestStore && (
                              <span className="text-xs bg-green-500 text-white px-2 py-0.5 rounded">
                                {t('priceSearch.best', '–õ—É—á—à–∞—è')}
                              </span>
                            )}
                          </h4>
                          <p className="text-sm text-gray-600">
                            {result.availability}
                          </p>
                        </div>
                        <p className="text-xl font-bold">
                          ${result.price.toFixed(2)}
                        </p>
                      </div>
                      {result.url && (
                        
                          href={result.url}
                          target="_blank"
                          rel="noopener noreferrer"
                          className="text-sm text-blue-600 hover:text-blue-800 flex items-center gap-1"
                        >
                          {t('priceSearch.openSite', '–û—Ç–∫—Ä—ã—Ç—å —Å–∞–π—Ç')}
                          <ExternalLink className="w-3 h-3" />
                        </a>
                      )}
                    </div>
                  ))}
                </div>
              </div>
            </div>
          )}
        </div>
        
        {/* Footer */}
        <div className="p-4 border-t flex justify-end">
          <button
            onClick={onClose}
            className="px-4 py-2 bg-gray-200 rounded-lg hover:bg-gray-300"
          >
            {t('common.close', '–ó–∞–∫—Ä—ã—Ç—å')}
          </button>
        </div>
      </div>
    </div>
  );
}
