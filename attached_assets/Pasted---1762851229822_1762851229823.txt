```typescript
import { db } from '../db/client';
import { calibrations, wallets, transactions } from '@shared/schema';
import { eq } from 'drizzle-orm';

/**
 * –û—Ç–∫–∞–ª–∏–±—Ä–æ–≤–∞—Ç—å –∫–æ—à–µ–ª—ë–∫
 * –°–æ–∑–¥–∞—ë—Ç —Ä–∞—Å—Ö–æ–¥ "–ù–µ—É—á—Ç—ë–Ω–Ω—ã–µ" –µ—Å–ª–∏ –±–∞–ª–∞–Ω—Å –º–µ–Ω—å—à–µ –æ–∂–∏–¥–∞–µ–º–æ–≥–æ
 */
export async function calibrateWallet(
  userId: number,
  walletId: number,
  actualBalance: number
): Promise {
  // 1. –ü–æ–ª—É—á–∏—Ç—å —Ç–µ–∫—É—â–∏–π –∫–æ—à–µ–ª—ë–∫
  const [wallet] = await db
    .select()
    .from(wallets)
    .where(eq(wallets.id, walletId))
    .limit(1);
  
  if (!wallet || wallet.userId !== userId) {
    throw new Error('Wallet not found');
  }
  
  const expectedBalance = Number(wallet.balance);
  const difference = actualBalance - expectedBalance;
  
  let transactionId: number | null = null;
  
  // 2. –ï—Å–ª–∏ —Ä–∞–∑–Ω–∏—Ü–∞ –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω–∞—è ‚Üí —Å–æ–∑–¥–∞—Ç—å —Ä–∞—Å—Ö–æ–¥
  if (difference < -0.01) {
    const [transaction] = await db
      .insert(transactions)
      .values({
        userId,
        type: 'expense',
        amount: Math.abs(difference),
        amountUsd: Math.abs(difference), // TODO: currency conversion
        description: `Unaccounted expenses (calibration of "${wallet.name}")`,
        category: 'Unaccounted',
        date: new Date().toISOString().split('T')[0],
        currency: wallet.currency || 'USD',
        source: 'calibration'
      })
      .returning();
    
    transactionId = transaction.id;
  }
  
  // 3. –û–±–Ω–æ–≤–∏—Ç—å –±–∞–ª–∞–Ω—Å –∫–æ—à–µ–ª—å–∫–∞
  await db
    .update(wallets)
    .set({ balance: actualBalance.toString() })
    .where(eq(wallets.id, walletId));
  
  // 4. –°–æ—Ö—Ä–∞–Ω–∏—Ç—å –∑–∞–ø–∏—Å—å –æ –∫–∞–ª–∏–±—Ä–æ–≤–∫–µ
  const [calibration] = await db
    .insert(calibrations)
    .values({
      userId,
      walletId,
      actualBalance: actualBalance.toString(),
      expectedBalance: expectedBalance.toString(),
      difference: difference.toString(),
      transactionId,
      date: new Date().toISOString().split('T')[0]
    })
    .returning();
  
  return {
    calibration,
    transactionCreated: transactionId !== null
  };
}

/**
 * –ü–æ–ª—É—á–∏—Ç—å –∏—Å—Ç–æ—Ä–∏—é –∫–∞–ª–∏–±—Ä–æ–≤–æ–∫ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
 */
export async function getCalibrationHistory(userId: number) {
  return await db
    .select({
      calibration: calibrations,
      wallet: wallets
    })
    .from(calibrations)
    .leftJoin(wallets, eq(calibrations.walletId, wallets.id))
    .where(eq(calibrations.userId, userId))
    .orderBy(calibrations.createdAt);
}
```
